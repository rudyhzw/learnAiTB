6 A2UI 消息类型

createSurface

创建 Surface

surfaceId: String
catalogId: String?
theme: Theme?
rootComponentId: String?

updateDataModel

更新数据模型

surfaceId: String
path: String
data: Any?

updateComponents

更新 UI 组件

surfaceId: String
components: List
  ├── id: String
  ├── component: Map
  ├── children: Map?
  └── events: Map?

deleteSurface

删除 Surface

surfaceId: String
